function seg_end = dubins_segment(seg_param, seg_init, seg_type)
    if( seg_type == 'L' ) 
        seg_end(1) = seg_init(1) + sin(seg_init(3)+seg_param) - sin(seg_init(3));
        seg_end(2) = seg_init(2) - cos(seg_init(3)+seg_param) + cos(seg_init(3));
        seg_end(3) = seg_init(3) + seg_param;
    elseif( seg_type == 'R' )
        seg_end(1) = seg_init(1) - sin(seg_init(3)-seg_param) + sin(seg_init(3));
        seg_end(2) = seg_init(2) + cos(seg_init(3)-seg_param) - cos(seg_init(3));
        seg_end(3) = seg_init(3) - seg_param;
    elseif( seg_type == 'S' ) 
        seg_end(1) = seg_init(1) + cos(seg_init(3)) * seg_param;
        seg_end(2) = seg_init(2) + sin(seg_init(3)) * seg_param;
        seg_end(3) = seg_init(3);
    end
end

function L = LSL(alpha,beta,d)
tmp0 = d + sin(alpha) - sin(beta);
p_squared = 2 + (d*d) -(2*cos(alpha - beta)) + (2*d*(sin(alpha) - sin(beta)));
if( p_squared < 0 )
    L = [inf inf inf inf];
else
    tmp1 = atan2( (cos(beta)-cos(alpha)), tmp0 );
    t = mod((-alpha + tmp1 ), 2*pi);
    p = sqrt( p_squared );
    q = mod((beta - tmp1 ), 2*pi);
    L=[t+p+q t p q];
end

end

public static class PathSegment implements LerpPath<Vector2dc>{
        private static final int SEGMENT_PRECISION = 10; // 每个段的插值精度

        public final SegmentType type;
        public final double length;  // 段长度（圆弧为弧度，直线为距离）
        public final Vector2dc startPoint;
        public final Vector2dc endPoint;
        public final Vector2dc center; // 圆心（仅圆弧有效）
        public final double radius;   // 转弯半径（仅圆弧有效）
        public final double startAngle; // 起始角度（仅圆弧有效）
        public final double endAngle;   // 结束角度（仅圆弧有效）

        public PathSegment(SegmentType type, double length,
                           Vector2d startPoint, Vector2d endPoint,
                           Vector2d center, double radius,
                           double startAngle, double endAngle) {
            this.type = type;
            this.length = length;
            this.startPoint = startPoint;
            this.endPoint = endPoint;
            this.center = center;
            this.radius = radius;
            this.startAngle = startAngle;
            this.endAngle = endAngle;
        }

        public PathSegment(SegmentType type,
                           Vector2d startPoint, Vector2d endPoint,
                           Vector2d center) {
            this.type = type;
            this.length = 0;
            this.startPoint = startPoint;
            this.endPoint = endPoint;
            this.center = center;
            this.radius = startPoint.distance(center);
            this.startAngle = 0;
            this.endAngle = 0;
        }

        @Override
        public int segments() {
            return SEGMENT_PRECISION;
        }

        @Override
        public Vector2dc lerp(int seg) {
            double t = (double) seg / SEGMENT_PRECISION; // 0 - 9
            if (type == SegmentType.STRAIGHT) {
                // 直线段插值

                return new Vector2d(
                        startPoint.x() + t * (endPoint.x() - startPoint.x()),
                        startPoint.y() + t * (endPoint.y() - startPoint.y())
                );
            } else {
                // 圆弧插值

                // 计算起点和终点相对于圆心的向量
                Vector2d startVec = new Vector2d(startPoint).sub(center);
                Vector2d endVec = new Vector2d(endPoint).sub(center);

                // 计算起点和终点的径向角度
                double currentRadialAngle = currentRadialAngle(startVec, endVec, t);

                // 计算当前点坐标
                double x = center.x() + radius * Math.cos(currentRadialAngle);
                double y = center.y() + radius * Math.sin(currentRadialAngle);
                return new Vector2d(x, y);
            }
        }

        private double currentRadialAngle(Vector2dc startVec, Vector2dc endVec, double t) {
            double startRadialAngle = Math.atan2(startVec.y(), startVec.x());
            double endRadialAngle = Math.atan2(endVec.y(), endVec.x());

            // 处理角度环绕问题
            double angularDistance;
            if (type == SegmentType.LEFT_TURN) {
                // 左转（逆时针）
                if (endRadialAngle < startRadialAngle) {
                    angularDistance = endRadialAngle - startRadialAngle + 2 * Math.PI;
                }else{
                    angularDistance = endRadialAngle - startRadialAngle;
                }

            } else {
                // 右转（顺时针）
                if (endRadialAngle < startRadialAngle) {
                    angularDistance = endRadialAngle - startRadialAngle;
                }else{
                    angularDistance = endRadialAngle - startRadialAngle - 2 * Math.PI;
                }

            }

            // 计算当前插值角度

            return startRadialAngle + t * angularDistance;
        }
    }

    // Dubins路径结果
    public static class DubinsPathResult implements LerpPath<Vector2d>{
        private static final int SEG_PER_SEGMENT = 10; // 每个路径段的插值点数

        public final PathType pathType;
        public final double totalLength;
        public final List<PathSegment> segments = new ArrayList<>();

        public DubinsPathResult(PathType pathType, double totalLength) {
            this.pathType = pathType;
            this.totalLength = totalLength;
        }

        @Override
        public int segments() {
            return 3 * SEG_PER_SEGMENT;
        }

        @Override
        public Vector2d lerp(int seg) {
            int segmentIndex = seg / SEG_PER_SEGMENT;
            int segmentOffset = seg % SEG_PER_SEGMENT;

            if (segmentIndex < segments.size()) {
                PathSegment segment = segments.get(segmentIndex);
                Vector2dc point = segment.lerp(segmentOffset);
                return new Vector2d(point.x(), point.y());
            } else {
                PathSegment lastSegment = segments.get(segments.size() - 1);
                Vector2dc lastPoint = lastSegment.endPoint;
                return new Vector2d(lastPoint.x(), lastPoint.y());
            }
        }
    }
